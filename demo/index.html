tiny-datepicker 演示

tiny-datepicker 是一个用原生的 JavaScript 实现的日期选择器的插件，无依赖，非常轻小。

下载 1.2.4

基本用法

默认 
            
<!-- HTML -->
<template>
  <input class="input-control" type="text" id="baseUsage"/>
</template>

<!-- JavaScript -->
<script>
  tinyDatePicker.init('#baseUsage');
</script>

组件用法

input组件 
            
<!-- HTML -->
<template>
  <div class="input-group">
    <i class="icon icon-date">
    <input class="input-control" type="text" id="componentUsage"/>
  </div>
</template>

<!-- JavaScript -->
<script>
  tinyDatePicker.init('.input-group', {
    clearable: true // 显示清除按钮
  });
</script>

国际化

i18n 
            
<!-- HTML -->
<template>
  <input class="input-control" type="text" id="i18nUsage"/>
</template>

<!-- JavaScript -->
<script src="${yourPath}/dist/locales/tiny-datepicker.en.min.js"></script>
<script>
  tinyDatePicker.init('#i18nUsage', {
    clearable: true, // show clear button
    lang: 'en' // select english for language
  });
</script>

禁用日期

禁用今天以后的日期 
            
<!-- HTML -->
<template>
  <input class="input-control" type="text" id="disableDateUsage"/>
</template>

<!-- JavaScript -->
<script>
  tinyDatePicker.init('#disableDateUsage', {
    disabledDate: function (date) { // 参数 date 是日期选择器中的某一个日期
      return date.getTime() > new Date().getTime();
    }
  });
</script>
